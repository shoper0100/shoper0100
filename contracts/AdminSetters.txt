    
    // ============ ADMIN CONFIGURATION FUNCTIONS ============
    
    function setDirectRequired(uint _newRequired) external {
        require(msg.sender == owner, \"Only owner\");
        require(_newRequired > 0 && _newRequired <= 10, \"Invalid value\");
        directRequired = _newRequired;
    }
    
    function setSponsorCommission(uint _newPercent) external {
        require(msg.sender == owner, \"Only owner\");
        require(_newPercent <= 20, \"Max 20%\");
        sponsorCommission = _newPercent;
    }
    
    function setMinSponsorLevel(uint _newLevel) external {
        require(msg.sender == owner, \"Only owner\");
        require(_newLevel > 0 && _newLevel <= 13, \"Invalid level\");
        minSponsorLevel = _newLevel;
    }
    
    function getDirectRequired() external view returns(uint) {
        return directRequired;
    }
    
    function getSponsorCommission() external view returns(uint) {
        return sponsorCommission;
    }
    
    function getMinSponsorLevel() external view returns(uint) {
        return minSponsorLevel;
    }
    
    function getOwner() external view returns(address) {
        return owner;
    }
