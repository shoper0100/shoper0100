    
    // Set BNB price in USD (for reference/calculation)
    function setBnbPrice(uint _priceInUSD) external {
        require(msg.sender == owner, \"Only owner\");
        require(_priceInUSD > 0, \"Price must be > 0\");
        bnbPriceInUSD = _priceInUSD;
    }
    
    // Batch update all level costs based on USD targets
    // _usdAmounts: array of 13 USD amounts for levels
    function batchUpdateLevels(uint[13] memory _usdAmounts) external {
        require(msg.sender == owner, \"Only owner\");
        require(bnbPriceInUSD > 0, \"Set BNB price first\");
        
        for(uint i=0; i<13; i++) {
            // Convert USD to BNB: (USD * 1e18) / BNB_price
            levels[i] = (_usdAmounts[i] * 1e18) / bnbPriceInUSD;
        }
    }
    
    // Get current BNB price
    function getBnbPrice() external view returns(uint) {
        return bnbPriceInUSD;
    }
